------EQUI-JOIN-------

SELECT PRODUCTNAME,PRICE,CATEGORYNAME,DESCRIPTION
FROM CATEGORY JOIN PRODUCTS
ON(CATEGORY.CATEGORYID = PRODUCTS.CATEGORYID)

-------OR-------

SELECT P.PRODUCTNAME,P.PRICE,C.CATEGORYNAME,C.DESCRIPTION
FROM CATEGORY C JOIN PRODUCTS P
ON(C.CATEGORYID = P.CATEGORYID)

-----OUTER JOIN------
-----LEFT OUTER JOIN---------

SELECT P.PRODUCTNAME,P.PRICE,C.CATEGORYNAME,C.DESCRIPTION
FROM CATEGORY C left JOIN PRODUCTS P
ON(C.CATEGORYID = P.CATEGORYID)

-----RIGHT OUTER JOIN--------

SELECT P.PRODUCTNAME,P.PRICE,C.CATEGORYNAME,C.DESCRIPTION
FROM CATEGORY C RIGHT JOIN PRODUCTS P
ON(C.CATEGORYID = P.CATEGORYID)

-----FULL OUTER JOIN---------

SELECT P.PRODUCTNAME,P.PRICE,C.CATEGORYNAME,C.DESCRIPTION
FROM CATEGORY C FULL JOIN PRODUCTS P
ON(C.CATEGORYID = P.CATEGORYID)
